from flask import Flask, render_template, request, redirect, url_for app = Flask(__name__) todos = [] @app.route('/') def index(): return render_template('index.html') @app.route('/todos') def todo_list(): return render_template('todos.html', todos=todos) @app.route('/add', methods=['POST']) def add(): todo = request.form.get('todo') todos.append(todo) return redirect(url_for('index')) @app.route('/delete/', methods=['GET']) def delete(index): if index < len(todos): del todos[index] return redirect(url_for('todo_list')) if __name__ == '__main__': app.run()
